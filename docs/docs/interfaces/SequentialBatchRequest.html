Interface SequentialBatchRequest

Sequential Batch Request configuration

Defines a single RPC request within a pair, with optional auto-chaining parameters.

Example

Basic request without chaining:
{
  method: 'getSignaturesForAddress',
  params: ['address', { limit: 10 }]
}

Example

Request with auto-chaining and expansion:
{
  method: 'getTransaction',
  inputFrom: 0,                      // Use output from request[0]
  inputPath: '[].signature',         // Extract all signatures
  expandArray: true,                 // Expand to N calls
  params: [null, { encoding: 'jsonParsed' }] // null will be replaced with signature
}
interface SequentialBatchRequest {
    expandArray?: boolean;
    id?: string | number;
    inputFrom?: number;
    inputPath?: string;
    method: string;
    params?: any[];
    targetParam?: number;
}
